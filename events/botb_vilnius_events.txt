    type = character_event
    title = vilnius.0001.t
    desc = {

                title:k_lithuania.holder = this
            }
            desc = vilnius.0001.kingly_comment           
        }
    }
    theme = hunting

    left_portrait = {
        character = root
        camera = camera_center_large
        animation = shock 
    }


    option = { # Consult priest - starts Vilnius event chain
        name = vilnius.0001.a

            base = 0

                add = 100
                has_trait = cynical
            }
        }
    }

}
