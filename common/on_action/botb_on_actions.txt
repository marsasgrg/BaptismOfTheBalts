dreams_of_the_wolf = {
    trigger = {
        involved_activity ?= { has_activity_type = activity_hunt }
        culture = culture:lithuanian
        title:c_kernave.holder = this
        title:b_vilnius.title_province = {
            has_holding = no
        }
        religion = religion:baltic_religion
        highest_held_title_tier >= tier_duchy
        exists = cp:councillor_court_chaplain
        OR = {
            is_independent_ruler = yes
            highest_held_title_tier >= tier_kingdom
        }
        NOT = {
            OR = {
                has_character_flag = a_dream_of_the_wolf_begins
                has_character_flag = a_dream_of_the_wolf
                has_character_flag = rejected_wolf_dream

            }
        }
    }
    effect = {
        
        random = {
            chance = 10
            trigger_event = {
                id = vilnius.0001
                days = 1
            } 
        }
    
    
    }
}
